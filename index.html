Sinkronisasi Google Spreadsheet

Simpan seluruh data progres ibadah Anda dengan aman ke Google Spreadsheet secara terpusat.

Skrip Kode backend (Code.gs)

Salin seluruh kode di bawah ini, buka script.google.com, tempelkan, buat Spreadsheet baru lalu hubungkan dengan mempublikasikannya sebagai Web App dengan akses untuk "Anyone". 

/*
 * ==========================================
 * Code.gs - Google Apps Script Backend Hijrah+
 * ==========================================
 */
function doGet(e) {
  // Cegah error jika dijalankan langsung di editor tanpa parameter
  if (!e || !e.parameter) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "success", 
      message: "Web App Hijrah+ aktif! Silakan hubungkan dari dashboard."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  var action = e.parameter.action;
  var sheetId = e.parameter.sheetId;
  
  var ss;
  try {
    if (sheetId) {
      ss = SpreadsheetApp.openById(sheetId);
    } else {
      ss = SpreadsheetApp.getActiveSpreadsheet();
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error", 
      message: "Gagal membuka Spreadsheet: " + err.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  if (!ss) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error", 
      message: "ID Spreadsheet tidak valid atau tidak terdeteksi."
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  setupSheets(ss);
  
  if (action === "get") {
    var payload = {
      status: "success",
      sholat: getSheetData(ss, "Sholat"),
      habit: getSheetData(ss, "Habit"),
      todo: getSheetData(ss, "Todo"),
      journal: getSheetData(ss, "Journal"),
      mood: getSheetData(ss, "Mood")
    };
    return ContentService.createTextOutput(JSON.stringify(payload)).setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput(JSON.stringify({status: "error", message: "Aksi tidak valid"})).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService.createTextOutput(JSON.stringify({
        status: "error", 
        message: "Data posting kosong atau tidak valid."
      })).setMimeType(ContentService.MimeType.JSON);
    }

    var postData = JSON.parse(e.postData.contents);
    var sheetId = postData.sheetId;
    var action = postData.action;
    var data = postData.data;
    
    var ss;
    if (sheetId) {
      ss = SpreadsheetApp.openById(sheetId);
    } else {
      ss = SpreadsheetApp.getActiveSpreadsheet();
    }
    
    if (!ss) {
      return ContentService.createTextOutput(JSON.stringify({
        status: "error", 
        message: "Spreadsheet tidak terdeteksi atau tidak valid."
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    setupSheets(ss);
    
    if (action === "sync") {
      syncTableData(ss, "Sholat", data.sholat || []);
      syncTableData(ss, "Habit", data.habit || []);
      syncTableData(ss, "Todo", data.todo || []);
      syncTableData(ss, "Journal", data.journal || []);
      syncTableData(ss, "Mood", data.mood || []);
      
      return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Data berhasil disinkronkan!"})).setMimeType(ContentService.MimeType.JSON);
    }
    
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: "Aksi tidak didukung"})).setMimeType(ContentService.MimeType.JSON);
    
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function setupSheets(ss) {
  if (!ss) {
    ss = SpreadsheetApp.getActiveSpreadsheet();
  }
  if (!ss) {
    throw new Error("Spreadsheet tidak valid atau tidak terhubung.");
  }
  var requiredSheets = ["Sholat", "Habit", "Todo", "Journal", "Mood"];
  requiredSheets.forEach(function(name) {
    var sheet = ss.getSheetByName(name);
    if (!sheet) {
      ss.insertSheet(name);
    }
  });
}

function getSheetData(ss, sheetName) {
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];
  var values = sheet.getDataRange().getValues();
  if (values.length <= 1) return [];
  
  var headers = values[0];
  var data = [];
  for (var i = 1; i < values.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = values[i][j];
    }
    data.push(obj);
  }
  return data;
}

function syncTableData(ss, sheetName, dataList) {
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return;
  sheet.clear();
  
  if (dataList.length === 0) {
    return;
  }
  
  var headers = Object.keys(dataList[0]);
  sheet.appendRow(headers);
  
  var rows = [];
  dataList.forEach(function(item) {
    var row = [];
    headers.forEach(function(header) {
      row.push(item[header] !== undefined ? item[header] : "");
    });
    rows.push(row);
  });
  
  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }
}
